Object-Oriented Modeling


from myhdl import *

def trigger(event):
    event.next = not event

class queue:
    def __init__(self):
       self.l = []
       self.sync = Signal(0)
       self.item = None
    def put(self, item):
       self.l.append(item)
       trigger(self.sync)
    def get(self):
       if not self.l:
          yield self.sync
       self.item = self.l.pop(0)
