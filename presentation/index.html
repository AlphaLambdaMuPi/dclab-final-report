Object-Oriented


class Button(init: Boolean) extends Bundle {
  val io = Reg(Bool(init))
  when (io) {
    io := Bool(init)
  }
  def push() = {
    io := Bool(!init)
  }
}

class Foo extends Module {
  val button = new Button(false)

Testing


class CoreTest(c: MyModule) extends Tester(c) {
  val ans = SOME_NUMBER
  poke(c.io.start, 1)
  step(1)
  poke(c.io.start, 0)
  while (!peek(c.io.done)) {
    step(1)
    peek(c.io.out)
  }
  expect(c.io.out, ans)
}
